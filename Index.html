Setup Guide
Step 1 — Go to sheets.google.com, create a new sheet named Cycle Tracker
Step 2 — Row 1 headers: A=StartDate, B=EndDate, C=PeriodDays, D=CycleLength, E=Notes
Step 3 — Extensions → Apps Script → paste the script below → Save
Step 4 — Deploy → New Deployment → Web App → Execute as Me → Anyone → Deploy
Step 5 — Copy the URL → paste above → Save & Test

function doGet(e) { ``` var ss = SpreadsheetApp.getActiveSpreadsheet(); var sh = ss.getSheetByName(“Cycle Tracker”); var action = e.parameter.action; if (action === “get”) { var data = sh.getDataRange().getValues(); var rows = []; for (var i = 1; i < data.length; i++) { var cell = data[i][0]; if (!cell || cell.toString().trim() === “”) continue; var startStr = cell instanceof Date ? Utilities.formatDate(cell, Session.getScriptTimeZone(), “yyyy-MM-dd”) : cell.toString().trim(); var endCell = data[i][1]; var endStr = endCell ? (endCell instanceof Date ? Utilities.formatDate(endCell, Session.getScriptTimeZone(), “yyyy-MM-dd”) : endCell.toString().trim()) : “”; rows.push({ startDate: startStr, endDate: endStr, periodDays: data[i][2] || “”, cycleLength: data[i][3] || “”, notes: data[i][4] || “” }); } return ContentService .createTextOutput(JSON.stringify({ok:true,data:rows})) .setMimeType(ContentService.MimeType.JSON); } if (action === “add”) { sh.appendRow([ e.parameter.startDate, e.parameter.endDate, e.parameter.periodDays, e.parameter.cycleLength, e.parameter.notes ]); return ContentService .createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } if (action === “addNote”) { var ns = ss.getSheetByName(“Notes”); if (!ns) { ns = ss.insertSheet(“Notes”); ns.appendRow([“Date”,“CycleDay”,“Phase”,“Note”]); } ns.appendRow([ e.parameter.date, e.parameter.day, e.parameter.phase, e.parameter.note ]); return ContentService .createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } return ContentService .createTextOutput(JSON.stringify({ok:false,error:“Unknown action”})) .setMimeType(ContentService.MimeType.JSON); }
